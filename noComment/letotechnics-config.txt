server {


	root /var/www/letotechnics.ru;

	index index.php index.html index.htm index.nginx-debian.html;

	server_name letotechnics.ru;

error_log /var/logs/letotechnics.error.log error;

	location / {
		try_files $uri $uri/ =404;
	}

	location ~ \.php$ {
		include snippets/fastcgi-php.conf;
		fastcgi_pass unix:/var/run/php/php7.2-fpm.sock;
	}

	location ~ /\.ht {
		deny all;
	}

    listen [::]:443 ssl ipv6only=on; 
    listen 443 ssl; 
    ssl_certificate /etc/letsencrypt/live/letotechnics.ru/fullchain.pem; 
    ssl_certificate_key /etc/letsencrypt/live/letotechnics.ru/privkey.pem; 
    include /etc/letsencrypt/options-ssl-nginx.conf; 
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; 

}
server {
    if ($host = letotechnics.ru) {
        return 301 https://$host$request_uri;
    } 


	listen 80;
	listen [::]:80;

	server_name letotechnics.ru;
    return 404; 


}
